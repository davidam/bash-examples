# Print a replacement in a file
sed -e 's/ABRIL/MAYO/g' | grep -i MAYO

# Execute a replacement in a file
sed -i 's/ABRIL/MAYO/g' femeninos.txt

# Print only lines from 2 to 6
yes | head -9 | cat -n | sed -n -e '/2/,/6/ p'

# Remove comments in file
sed -e '/^#/d' /etc/services

# Escaping slashes
sed 's/\/bin\/bash/\/bin\/csh/' /etc/passwd  | grep csh

# Replacing only some lines
sed "2,3s/test/another test/" test.txt

# Remove tags from html file
sed -e 's/<.*>//g' geeks.html

# Replacing from a file
sed -f replace.txt < a.txt
